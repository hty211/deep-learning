# 序列模型架构

## 1. Seq2Seq模型

序列到序列模型用于将一个序列转换为另一个序列。

### 1.1 Encoder-Decoder架构

```python
import torch
import torch.nn as nn

class Encoder(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size, num_layers=2):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        self.rnn = nn.LSTM(embed_size, hidden_size, num_layers, batch_first=True)
    
    def forward(self, x):
        embedded = self.embedding(x)
        outputs, (hidden, cell) = self.rnn(embedded)
        return outputs, hidden, cell

class Decoder(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size, num_layers=2):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        self.rnn = nn.LSTM(embed_size, hidden_size, num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_size, vocab_size)
    
    def forward(self, x, hidden, cell):
        embedded = self.embedding(x)
        output, (hidden, cell) = self.rnn(embedded, (hidden, cell))
        prediction = self.fc(output)
        return prediction, hidden, cell

class Seq2Seq(nn.Module):
    def __init__(self, encoder, decoder, device):
        super().__init__()
        self.encoder = encoder
        self.decoder = decoder
        self.device = device
    
    def forward(self, src, trg, teacher_forcing_ratio=0.5):
        batch_size = src.shape[0]
        trg_len = trg.shape[1]
        trg_vocab_size = self.decoder.fc.out_features
        
        outputs = torch.zeros(batch_size, trg_len, trg_vocab_size).to(self.device)
        
        _, hidden, cell = self.encoder(src)
        
        input = trg[:, 0].unsqueeze(1)
        
        for t in range(1, trg_len):
            output, hidden, cell = self.decoder(input, hidden, cell)
            outputs[:, t] = output.squeeze(1)
            
            teacher_force = torch.rand(1) < teacher_forcing_ratio
            top1 = output.argmax(2)
            input = trg[:, t].unsqueeze(1) if teacher_force else top1
        
        return outputs

vocab_size = 1000
embed_size = 128
hidden_size = 256
num_layers = 2
device = torch.device('cpu')

encoder = Encoder(vocab_size, embed_size, hidden_size, num_layers)
decoder = Decoder(vocab_size, embed_size, hidden_size, num_layers)
seq2seq = Seq2Seq(encoder, decoder, device)

src = torch.randint(0, vocab_size, (4, 10))
trg = torch.randint(0, vocab_size, (4, 12))

output = seq2seq(src, trg)
print(f"源序列形状: {src.shape}")
print(f"目标序列形状: {trg.shape}")
print(f"输出形状: {output.shape}")
```

## 2. 注意力机制

### 2.1 基本注意力

```python
class Attention(nn.Module):
    def __init__(self, hidden_size):
        super().__init__()
        self.attn = nn.Linear(hidden_size * 2, hidden_size)
        self.v = nn.Linear(hidden_size, 1, bias=False)
    
    def forward(self, hidden, encoder_outputs):
        src_len = encoder_outputs.shape[1]
        
        hidden = hidden.unsqueeze(1).repeat(1, src_len, 1)
        
        energy = torch.tanh(self.attn(torch.cat((hidden, encoder_outputs), dim=2)))
        
        attention = self.v(energy).squeeze(2)
        
        return torch.softmax(attention, dim=1)

class AttnDecoder(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size, num_layers=2):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        self.attention = Attention(hidden_size)
        self.rnn = nn.LSTM(hidden_size + embed_size, hidden_size, num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_size * 2, vocab_size)
    
    def forward(self, x, hidden, cell, encoder_outputs):
        embedded = self.embedding(x)
        
        attn_weights = self.attention(hidden[-1], encoder_outputs)
        attn_weights = attn_weights.unsqueeze(1)
        
        context = torch.bmm(attn_weights, encoder_outputs)
        
        rnn_input = torch.cat((embedded, context), dim=2)
        
        output, (hidden, cell) = self.rnn(rnn_input, (hidden, cell))
        
        output = torch.cat((output.squeeze(1), context.squeeze(1)), dim=1)
        prediction = self.fc(output)
        
        return prediction, hidden, cell, attn_weights

attn_decoder = AttnDecoder(vocab_size, embed_size, hidden_size)
encoder_outputs = torch.randn(4, 10, hidden_size)
hidden = torch.randn(num_layers, 4, hidden_size)
cell = torch.randn(num_layers, 4, hidden_size)
x = torch.randint(0, vocab_size, (4, 1))

pred, h, c, attn = attn_decoder(x, hidden, cell, encoder_outputs)
print(f"注意力权重形状: {attn.shape}")
print(f"预测形状: {pred.shape}")
```

### 2.2 自注意力

```python
class SelfAttention(nn.Module):
    def __init__(self, hidden_size, num_heads=8):
        super().__init__()
        
        self.num_heads = num_heads
        self.head_dim = hidden_size // num_heads
        
        self.q_linear = nn.Linear(hidden_size, hidden_size)
        self.k_linear = nn.Linear(hidden_size, hidden_size)
        self.v_linear = nn.Linear(hidden_size, hidden_size)
        
        self.out_linear = nn.Linear(hidden_size, hidden_size)
    
    def forward(self, x, mask=None):
        batch_size = x.shape[0]
        
        Q = self.q_linear(x).view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        K = self.k_linear(x).view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        V = self.v_linear(x).view(batch_size, -1, self.num_heads, self.head_dim).transpose(1, 2)
        
        scores = torch.matmul(Q, K.transpose(-2, -1)) / (self.head_dim ** 0.5)
        
        if mask is not None:
            scores = scores.masked_fill(mask == 0, -1e9)
        
        attn_weights = torch.softmax(scores, dim=-1)
        
        out = torch.matmul(attn_weights, V)
        
        out = out.transpose(1, 2).contiguous().view(batch_size, -1, self.num_heads * self.head_dim)
        
        return self.out_linear(out)

self_attn = SelfAttention(hidden_size=256, num_heads=8)
x = torch.randn(4, 10, 256)
out = self_attn(x)
print(f"自注意力输入: {x.shape}")
print(f"自注意力输出: {out.shape}")
```

## 3. 双向RNN

```python
class BiLSTMClassifier(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size, output_size, num_layers=2):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        
        self.lstm = nn.LSTM(
            embed_size, hidden_size, num_layers,
            batch_first=True, bidirectional=True, dropout=0.3
        )
        
        self.fc = nn.Sequential(
            nn.Linear(hidden_size * 2, hidden_size),
            nn.ReLU(),
            nn.Dropout(0.5),
            nn.Linear(hidden_size, output_size)
        )
    
    def forward(self, x):
        embedded = self.embedding(x)
        
        lstm_out, _ = self.lstm(embedded)
        
        avg_pool = lstm_out.mean(dim=1)
        max_pool = lstm_out.max(dim=1)[0]
        
        combined = torch.cat([avg_pool, max_pool], dim=1)
        
        return self.fc(lstm_out[:, -1, :])

model = BiLSTMClassifier(vocab_size=10000, embed_size=128, 
                         hidden_size=256, output_size=2)
x = torch.randint(0, 10000, (4, 20))
output = model(x)
```

## 4. 多层RNN

```python
class MultiLayerRNN(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size, output_size, num_layers=3):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        
        self.rnn_layers = nn.ModuleList([
            nn.LSTM(embed_size if i == 0 else hidden_size, 
                   hidden_size, batch_first=True, dropout=0.2)
            for i in range(num_layers)
        ])
        
        self.fc = nn.Linear(hidden_size, output_size)
    
    def forward(self, x):
        embedded = self.embedding(x)
        
        output = embedded
        for rnn in self.rnn_layers:
            output, _ = rnn(output)
        
        return self.fc(output[:, -1, :])

model = MultiLayerRNN(vocab_size=10000, embed_size=128, 
                      hidden_size=256, output_size=2, num_layers=3)
x = torch.randint(0, 10000, (4, 20))
output = model(x)
print(f"多层RNN输出: {output.shape}")
```

## 5. 带有Dropout的RNN

```python
class RNNWithDropout(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size, output_size, dropout=0.5):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        self.dropout = nn.Dropout(dropout)
        
        self.lstm = nn.LSTM(embed_size, hidden_size, num_layers=2,
                           batch_first=True, dropout=dropout, bidirectional=True)
        
        self.fc = nn.Sequential(
            nn.Linear(hidden_size * 2, hidden_size),
            nn.ReLU(),
            nn.Dropout(dropout),
            nn.Linear(hidden_size, output_size)
        )
    
    def forward(self, x):
        embedded = self.dropout(self.embedding(x))
        lstm_out, _ = self.lstm(embedded)
        return self.fc(lstm_out[:, -1, :])

model = RNNWithDropout(vocab_size=10000, embed_size=128, 
                       hidden_size=256, output_size=2, dropout=0.5)

model.train()
output_train = model(x)
print(f"训练模式输出: {output_train.shape}")

model.eval()
output_eval = model(x)
print(f"评估模式输出: {output_eval.shape}")
```

## 6. 序列到序列变体

```python
class LSTMWithAttention(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size, output_size):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        
        self.lstm = nn.LSTM(embed_size, hidden_size, batch_first=True, bidirectional=True)
        
        self.attention = nn.Sequential(
            nn.Linear(hidden_size * 2, hidden_size),
            nn.Tanh(),
            nn.Linear(hidden_size, 1)
        )
        
        self.fc = nn.Linear(hidden_size * 2, output_size)
    
    def forward(self, x):
        embedded = self.embedding(x)
        lstm_out, _ = self.lstm(embedded)
        
        attn_weights = self.attention(lstm_out)
        attn_weights = torch.softmax(attn_weights, dim=1)
        
        context = torch.sum(lstm_out * attn_weights, dim=1)
        
        return self.fc(context)

model = LSTMWithAttention(vocab_size=10000, embed_size=128, 
                          hidden_size=256, output_size=2)
x = torch.randint(0, 10000, (4, 20))
output = model(x)
print(f"带注意力的LSTM输出: {output.shape}")
```

## 7. 总结

| 架构 | 特点 | 应用场景 |
|------|------|---------|
| Seq2Seq | 编码器-解码器 | 机器翻译、文本摘要 |
| 注意力机制 | 动态关注重要部分 | 长序列处理 |
| 双向RNN | 获取双向上下文 | 文本分类、命名实体识别 |
| 多层RNN | 学习更高层次特征 | 复杂序列任务 |
| 自注意力 | 全局依赖建模 | Transformer基础 |
