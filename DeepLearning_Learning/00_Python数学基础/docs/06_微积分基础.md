# 微积分基础

## 1. 导数

### 1.1 导数的定义

导数描述函数在某点的变化率：

f'(x) = lim(h→0) [f(x+h) - f(x)] / h

```python
import numpy as np

def numerical_derivative(f, x, h=1e-5):
    return (f(x + h) - f(x - h)) / (2 * h)

def f(x):
    return x**2

x = 3
derivative = numerical_derivative(f, x)
print(f"f(x) = x² 在 x=3 处的导数: {derivative}")
print(f"理论值: {2 * x}")
```

### 1.3 链式法则

如果 y = f(g(x))，则 dy/dx = f'(g(x)) * g'(x)

```python
def chain_rule_example():
    x = 2.0
    
    g = lambda x: x**2
    f = lambda u: np.exp(u)
    
    y = lambda x: f(g(x))
    
    dy_dx = numerical_derivative(y, x)
    
    dg_dx = numerical_derivative(g, x)
    df_du = numerical_derivative(f, g(x))
    chain_result = df_du * dg_dx
    
    print(f"链式法则验证:")
    print(f"y = exp(x²)")
    print(f"数值导数: {dy_dx:.6f}")
    print(f"链式法则: {chain_result:.6f}")
    print(f"理论值: {2*x*np.exp(x**2):.6f}")

chain_rule_example()
```

## 2. 偏导数

### 2.1 偏导数定义

对于多元函数 f(x₁, x₂, ..., xₙ)，偏导数是对其中一个变量的导数：

∂f/∂xᵢ

```python
def partial_derivative(f, x, i, h=1e-5):
    x_plus = x.copy()
    x_minus = x.copy()
    x_plus[i] += h
    x_minus[i] -= h
    return (f(x_plus) - f(x_minus)) / (2 * h)

def f_multi(x):
    return x[0]**2 + x[1]**2 + x[0]*x[1]

x = np.array([1.0, 2.0])

df_dx0 = partial_derivative(f_multi, x, 0)
df_dx1 = partial_derivative(f_multi, x, 1)

print(f"f(x0, x1) = x0² + x1² + x0*x1")
print(f"在点 (1, 2) 处:")
print(f"∂f/∂x0 = 2x0 + x1 = {2*1 + 2}, 数值: {df_dx0:.6f}")
print(f"∂f/∂x1 = 2x1 + x0 = {2*2 + 1}, 数值: {df_dx1:.6f}")
```

### 2.2 梯度

梯度是所有偏导数组成的向量：

∇f = (∂f/∂x₁, ∂f/∂x₂, ..., ∂f/∂xₙ)

```python
def gradient(f, x, h=1e-5):
    grad = np.zeros_like(x)
    for i in range(len(x)):
        grad[i] = partial_derivative(f, x, i, h)
    return grad

def f_multi(x):
    return x[0]**2 + x[1]**2

x = np.array([3.0, 4.0])
grad = gradient(f_multi, x)

print(f"f(x) = x0² + x1²")
print(f"在点 (3, 4) 处的梯度: {grad}")
print(f"理论梯度: [2*3, 2*4] = [{2*3}, {2*4}]")
```

## 3. 梯度下降

### 3.1 基本原理

梯度下降通过迭代更新参数来最小化函数：

x_new = x - α * ∇f(x)

```python
def gradient_descent(f, grad_f, x_init, learning_rate=0.1, max_iter=100, tol=1e-6):
    x = x_init.copy()
    history = [x.copy()]
    
    for i in range(max_iter):
        grad = grad_f(x)
        x_new = x - learning_rate * grad
        
        if np.linalg.norm(x_new - x) < tol:
            print(f"收敛于迭代 {i+1}")
            break
            
        x = x_new
        history.append(x.copy())
    
    return x, np.array(history)

def f_rosenbrock(x):
    return (1 - x[0])**2 + 100 * (x[1] - x[0]**2)**2

def grad_rosenbrock(x):
    grad = np.zeros(2)
    grad[0] = -2*(1 - x[0]) - 400*x[0]*(x[1] - x[0]**2)
    grad[1] = 200*(x[1] - x[0]**2)
    return grad

x_init = np.array([-1.0, 1.0])
x_min, history = gradient_descent(f_rosenbrock, grad_rosenbrock, x_init, 
                                   learning_rate=0.001, max_iter=10000)

print(f"初始点: {x_init}")
print(f"最小值点: {x_min}")
print(f"最小值: {f_rosenbrock(x_min):.6f}")
print(f"理论最小值点: [1, 1]")
```

## 4. 积分

### 4.1 定积分

∫[a,b] f(x) dx

```python
def numerical_integral(f, a, b, n=1000):
    x = np.linspace(a, b, n)
    dx = (b - a) / (n - 1)
    return np.sum(f(x)) * dx

def f(x):
    return x**2

a, b = 0, 1
integral = numerical_integral(f, a, b)
theoretical = (b**3 - a**3) / 3

print(f"∫₀¹ x² dx")
print(f"数值积分: {integral:.6f}")
print(f"理论值: {theoretical:.6f}")
```

## 5. 在深度学习中的应用

### 5.1 反向传播

```python
import numpy as np

class SimpleNN:
    def __init__(self, input_size, hidden_size, output_size):
        self.W1 = np.random.randn(input_size, hidden_size) * 0.01
        self.b1 = np.zeros((1, hidden_size))
        self.W2 = np.random.randn(hidden_size, output_size) * 0.01
        self.b2 = np.zeros((1, output_size))
    
    def sigmoid(self, x):
        return 1 / (1 + np.exp(-x))
    
    def sigmoid_derivative(self, x):
        return x * (1 - x)
    
    def forward(self, X):
        self.z1 = X @ self.W1 + self.b1
        self.a1 = self.sigmoid(self.z1)
        self.z2 = self.a1 @ self.W2 + self.b2
        self.a2 = self.sigmoid(self.z2)
        return self.a2
    
    def backward(self, X, y, learning_rate=0.1):
        m = X.shape[0]
        
        dz2 = self.a2 - y
        dW2 = (self.a1.T @ dz2) / m
        db2 = np.sum(dz2, axis=0, keepdims=True) / m
        
        da1 = dz2 @ self.W2.T
        dz1 = da1 * self.sigmoid_derivative(self.a1)
        dW1 = (X.T @ dz1) / m
        db1 = np.sum(dz1, axis=0, keepdims=True) / m
        
        self.W2 -= learning_rate * dW2
        self.b2 -= learning_rate * db2
        self.W1 -= learning_rate * dW1
        self.b1 -= learning_rate * db1
    
    def train(self, X, y, epochs=1000, learning_rate=0.1):
        for i in range(epochs):
            output = self.forward(X)
            self.backward(X, y, learning_rate)
            if i % 100 == 0:
                loss = np.mean((output - y)**2)
                print(f"Epoch {i}, Loss: {loss:.6f}")

X = np.array([[0, 0], [0, 1], [1, 0], [1, 1]])
y = np.array([[0], [1], [1], [0]])

nn = SimpleNN(2, 4, 1)
nn.train(X, y, epochs=2000, learning_rate=1)

print("\n预测结果:")
print(nn.forward(X))
```

### 5.2 激活函数的导数

```python
def relu(x):
    return np.maximum(0, x)

def relu_derivative(x):
    return (x > 0).astype(float)

def leaky_relu(x, alpha=0.01):
    return np.where(x > 0, x, alpha * x)

def leaky_relu_derivative(x, alpha=0.01):
    return np.where(x > 0, 1, alpha)

def tanh_derivative(x):
    return 1 - x**2

x = np.array([-2, -1, 0, 1, 2])
print(f"ReLU: {relu(x)}")
print(f"ReLU导数: {relu_derivative(x)}")
print(f"Leaky ReLU: {leaky_relu(x)}")
print(f"Leaky ReLU导数: {leaky_relu_derivative(x)}")
```

## 6. 重要概念总结

| 概念 | 定义 | 在DL中的应用 |
|------|------|--------------|
| 导数 | 函数变化率 | 梯度计算 |
| 偏导数 | 多元函数对一个变量的导数 | 参数更新 |
| 梯度 | 偏导数组成的向量 | 梯度下降 |
| 链式法则 | 复合函数求导 | 反向传播 |
| 梯度下降 | x = x - α∇f | 优化算法 |
