# 概率论基础

## 1. 基本概念

### 1.1 概率定义

概率是描述随机事件发生可能性的数值，取值范围为[0, 1]。

```python
import numpy as np

def coin_flip(n=1):
    return np.random.choice(['H', 'T'], size=n, p=[0.5, 0.5])

flips = coin_flip(1000)
p_head = np.sum(flips == 'H') / len(flips)
print(f"正面概率估计: {p_head}")
```

### 1.2 条件概率

P(A|B) = P(A∩B) / P(B)

```python
import pandas as pd

data = {
    'Disease': [1, 1, 0, 0, 1, 0, 1, 0, 0, 1],
    'Test': [1, 1, 1, 0, 1, 0, 0, 1, 0, 1]
}
df = pd.DataFrame(data)

p_disease = df['Disease'].mean()
p_positive = df['Test'].mean()

p_positive_given_disease = df[df['Disease'] == 1]['Test'].mean()

p_disease_given_positive = (p_positive_given_disease * p_disease) / p_positive

print(f"P(Disease): {p_disease}")
print(f"P(Positive|Disease): {p_positive_given_disease}")
print(f"P(Disease|Positive): {p_disease_given_positive}")
```

### 1.3 贝叶斯定理

P(A|B) = P(B|A) * P(A) / P(B)

```python
p_disease = 0.01
p_positive_given_disease = 0.99
p_positive_given_no_disease = 0.05

p_positive = p_positive_given_disease * p_disease + p_positive_given_no_disease * (1 - p_disease)

p_disease_given_positive = (p_positive_given_disease * p_disease) / p_positive

print(f"患病概率: {p_disease}")
print(f"检测阳性概率: {p_positive}")
print(f"阳性时真正患病的概率: {p_disease_given_positive}")
```

## 2. 概率分布

```python
p = 0.3
```

#### 二项分布

```python
n, p = 10, 0.5
```

### 2.2 连续分布

#### 正态分布（高斯分布）

```python
mu, sigma = 0, 1

print(f"样本均值: {samples.mean()}")
print(f"理论标准差: {sigma}")

import matplotlib.pyplot as plt

plt.plot(x, 1/(sigma * np.sqrt(2 * np.pi)) * np.exp(-(x - mu)**2 / (2 * sigma**2)))

plt.show()
```

## 3. 期望与方差

### 3.1 期望

E[X] = Σ x * P(x) （离散）
E[X] = ∫ x * f(x) dx （连续）

```python
x = np.array([1, 2, 3, 4, 5])
p = np.array([0.1, 0.2, 0.3, 0.25, 0.15])

expected_value = np.sum(x * p)
print(f"期望: {expected_value}")

samples = np.random.choice(x, size=10000, p=p)
print(f"样本均值: {samples.mean()}")
```

### 3.2 方差

Var(X) = E[(X - E[X])²] = E[X²] - (E[X])²

```python
x = np.array([1, 2, 3, 4, 5])
p = np.array([0.1, 0.2, 0.3, 0.25, 0.15])

E_X = np.sum(x * p)
E_X2 = np.sum(x**2 * p)
variance = E_X2 - E_X**2

print(f"期望: {E_X}")
print(f"方差: {variance}")
print(f"标准差: {np.sqrt(variance)}")

samples = np.random.choice(x, size=10000, p=p)
print(f"样本方差: {samples.var()}")
```

## 4. 常用概率函数

```python
from scipy import stats

mu, sigma = 0, 1
```

### 4.2 累积分布函数 (CDF)

```python
x = 0
cdf = stats.norm.cdf(x, mu, sigma)
print(f"CDF在x=0处的值: {cdf}")

x = 1.96
cdf = stats.norm.cdf(x, mu, sigma)
print(f"P(X < 1.96): {cdf}")
print(f"P(-1.96 < X < 1.96): {stats.norm.cdf(1.96) - stats.norm.cdf(-1.96)}")
```

## 5. 大数定律与中心极限定理

### 5.1 大数定律

```python
import matplotlib.pyplot as plt

p = 0.5
n_samples = 10000
samples = np.random.binomial(1, p, n_samples)

cumulative_mean = np.cumsum(samples) / np.arange(1, n_samples + 1)

plt.figure(figsize=(10, 6))
plt.plot(cumulative_mean)
plt.axhline(y=p, color='r', linestyle='--', label=f'真实概率 p={p}')
plt.xlabel('样本数')
plt.ylabel('累积均值')
plt.title('大数定律演示')
plt.legend()
plt.show()
```

### 5.2 中心极限定理

```python
import matplotlib.pyplot as plt

n_samples = 1000
sample_size = 30
population_mean = 3
population_std = 2

sample_means = [np.random.normal(population_mean, population_std, sample_size).mean() 
                for _ in range(n_samples)]

plt.figure(figsize=(10, 6))
plt.hist(sample_means, bins=30, density=True, alpha=0.7)

theoretical_std = population_std / np.sqrt(sample_size)
x = np.linspace(population_mean - 4*theoretical_std, population_mean + 4*theoretical_std, 100)
plt.plot(x, stats.norm.pdf(x, population_mean, theoretical_std), 'r-', linewidth=2)
plt.title('中心极限定理演示')
plt.xlabel('样本均值')
plt.ylabel('密度')
plt.show()

print(f"理论均值: {population_mean}")
print(f"样本均值分布的均值: {np.mean(sample_means)}")
print(f"理论标准误: {theoretical_std}")
print(f"样本均值分布的标准差: {np.std(sample_means)}")
```

## 6. 在深度学习中的应用

### 6.1 最大似然估计

```python
def mle_normal(data):
    mu = np.mean(data)
    sigma = np.std(data)
    return mu, sigma

data = np.random.normal(5, 2, 1000)
mu_hat, sigma_hat = mle_normal(data)
print(f"真实参数: mu=5, sigma=2")
print(f"估计参数: mu={mu_hat:.3f}, sigma={sigma_hat:.3f}")
```

### 6.2 交叉熵损失

```python
def cross_entropy_loss(y_true, y_pred):
    epsilon = 1e-15
    y_pred = np.clip(y_pred, epsilon, 1 - epsilon)
    return -np.mean(y_true * np.log(y_pred) + (1 - y_true) * np.log(1 - y_pred))

y_true = np.array([1, 0, 1, 1, 0])
y_pred = np.array([0.9, 0.1, 0.8, 0.7, 0.2])

loss = cross_entropy_loss(y_true, y_pred)
print(f"交叉熵损失: {loss}")
```

### 6.3 Softmax函数

```python
def softmax(x):
    exp_x = np.exp(x - np.max(x))
    return exp_x / np.sum(exp_x)

logits = np.array([2.0, 1.0, 0.1])
probs = softmax(logits)
print(f"Logits: {logits}")
print(f"概率分布: {probs}")
print(f"概率和: {probs.sum()}")
```

## 7. 重要概念总结

| 概念 | 定义 | 在DL中的应用 |
|------|------|--------------|
| 条件概率 | P(A\|B) = P(A∩B)/P(B) | 贝叶斯推断 |
| 贝叶斯定理 | P(A\|B) = P(B\|A)P(A)/P(B) | 模型推断 |
| 正态分布 | N(μ, σ²) | 权重初始化、噪声模型 |
| 期望 | E[X] | 损失函数期望 |
| 方差 | Var(X) | 模型不确定性 |
| 最大似然估计 | argmax P(data\|θ) | 参数估计 |
| 交叉熵 | -Σy*log(ŷ) | 分类损失函数 |
| Softmax | exp(x_i)/Σexp(x_j) | 多分类输出 |
