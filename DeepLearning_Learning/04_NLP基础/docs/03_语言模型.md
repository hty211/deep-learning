## 2. N-gram语言模型

### 2.1 基本原理

```python
from collections import defaultdict
import numpy as np

class NGramLM:
    def __init__(self, n=2):
        self.n = n
        self.ngram_counts = defaultdict(int)
        self.context_counts = defaultdict(int)
        self.vocab = set()
    
    def train(self, corpus):
        for sentence in corpus:
            tokens = ['<s>'] * (self.n - 1) + sentence.split() + ['</s>']
            
            for token in tokens:
                self.vocab.add(token)
            
            for i in range(len(tokens) - self.n + 1):
                ngram = tuple(tokens[i:i+self.n])
                context = tuple(tokens[i:i+self.n-1])
                
                self.ngram_counts[ngram] += 1
                self.context_counts[context] += 1
    
    def probability(self, word, context):
        ngram = context + (word,)
        context_tuple = context
        
        if self.context_counts[context_tuple] == 0:
            return 1 / len(self.vocab)
        
        return self.ngram_counts[ngram] / self.context_counts[context_tuple]
    
    def sentence_probability(self, sentence):
        tokens = ['<s>'] * (self.n - 1) + sentence.split() + ['</s>']
        prob = 0
        
        for i in range(self.n - 1, len(tokens)):
            context = tuple(tokens[i-self.n+1:i])
            word = tokens[i]
            p = self.probability(word, context)
            prob += np.log(p + 1e-10)
        
        return prob
    
    def generate(self, max_length=20):
        context = ['<s>'] * (self.n - 1)
        generated = []
        
        for _ in range(max_length):
            candidates = []
            for word in self.vocab:
                if word not in ['<s>', '</s>']:
                    p = self.probability(word, tuple(context))
                    candidates.append((word, p))
            
            if not candidates:
                break
            
            candidates.sort(key=lambda x: x[1], reverse=True)
            next_word = candidates[0][0]
            
            if next_word == '</s>':
                break
            
            generated.append(next_word)
            context = context[1:] + [next_word]
        
        return ' '.join(generated)

corpus = [
    "I love natural language processing",
    "I love deep learning",
    "natural language processing is fun",
    "deep learning is powerful"
]

print("训练语料:")
for sent in corpus:
    print(f"  {sent}")

bigram = NGramLM(n=2)
bigram.train(corpus)

print("\nBigram概率示例:")
print(f"  P(love | I) = {bigram.probability('love', ('I',)):.4f}")
print(f"  P(language | natural) = {bigram.probability('language', ('natural',)):.4f}")

print("\n句子概率:")
test_sentences = ["I love deep learning", "deep learning love I"]

    prob = bigram.sentence_probability(sent)
    print(f"  P('{sent}') = {prob:.4f}")
```

### 2.2 平滑技术

```python
class SmoothedNGramLM(NGramLM):
    def __init__(self, n=2, smoothing='laplace'):
        super().__init__(n)
        self.smoothing = smoothing
    
    def probability(self, word, context):
        ngram = context + (word,)
        context_tuple = context
        
        if self.smoothing == 'laplace':
            count = self.ngram_counts[ngram] + 1
            total = self.context_counts[context_tuple] + len(self.vocab)
            return count / total
        
        elif self.smoothing == 'add_k':
            k = 0.5
            count = self.ngram_counts[ngram] + k
            total = self.context_counts[context_tuple] + k * len(self.vocab)
            return count / total
        
        return super().probability(word, context)

smoothed_lm = SmoothedNGramLM(n=2, smoothing='laplace')
smoothed_lm.train(corpus)
```

### 2.3 困惑度

```python
def perplexity(lm, test_corpus):
    total_log_prob = 0
    total_words = 0
    
    for sentence in test_corpus:
        tokens = sentence.split()
        total_words += len(tokens)
        total_log_prob += lm.sentence_probability(sentence)
    
    avg_log_prob = total_log_prob / total_words
    perplexity = np.exp(-avg_log_prob)
    
    return perplexity

test_corpus = [
    "I love natural language",
    "deep learning is fun"
]

train_perplexity = perplexity(bigram, corpus)
test_perplexity = perplexity(bigram, test_corpus)

print(f"训练集困惑度: {train_perplexity:.4f}")
print(f"测试集困惑度: {test_perplexity:.4f}")

print("\n困惑度解释:")
print("  - 困惑度越低，模型越好")
print("  - 困惑度 = 2^(交叉熵)")
print("  - 表示模型对下一个词的平均分支因子")
```

## 3. 神经语言模型

### 3.1 前馈神经语言模型

```python
import torch
import torch.nn as nn

class FeedForwardLM(nn.Module):
    def __init__(self, vocab_size, embedding_dim, hidden_dim, context_size):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embedding_dim)
        self.fc1 = nn.Linear(context_size * embedding_dim, hidden_dim)
        self.fc2 = nn.Linear(hidden_dim, vocab_size)
        
    def forward(self, x):
        embedded = self.embedding(x)
        embedded = embedded.view(embedded.size(0), -1)
        
        hidden = torch.relu(self.fc1(embedded))
        output = self.fc2(hidden)
        
        return output

vocab_size = 1000
embedding_dim = 50
hidden_dim = 100
context_size = 3

ff_lm = FeedForwardLM(vocab_size, embedding_dim, hidden_dim, context_size)

context = torch.randint(0, vocab_size, (2, context_size))
output = ff_lm(context)

print(f"前馈语言模型:")
print(f"  输入(上下文): {context.shape}")
print(f"  输出(词表概率): {output.shape}")
```

### 3.2 RNN语言模型

```python
class RNNLM(nn.Module):
    def __init__(self, vocab_size, embedding_dim, hidden_dim, num_layers=2):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embedding_dim)
        self.rnn = nn.LSTM(embedding_dim, hidden_dim, num_layers, 
                          batch_first=True, dropout=0.2)
        self.fc = nn.Linear(hidden_dim, vocab_size)
    
    def forward(self, x, hidden=None):
        embedded = self.embedding(x)
        output, hidden = self.rnn(embedded, hidden)
        output = self.fc(output)
        return output, hidden
    
    def generate(self, start_token, max_length=50, temperature=1.0):
        self.eval()
        generated = [start_token]
        hidden = None
        
        x = torch.tensor([[start_token]])
        
        with torch.no_grad():
            for _ in range(max_length):
                output, hidden = self(x, hidden)
                
                probs = torch.softmax(output[0, -1] / temperature, dim=0)
                next_token = torch.multinomial(probs, 1).item()
                
                generated.append(next_token)
                x = torch.tensor([[next_token]])
                
                if next_token == 2:
                    break
        
        return generated

rnn_lm = RNNLM(vocab_size, embedding_dim=128, hidden_dim=256, num_layers=2)

x = torch.randint(0, vocab_size, (2, 20))
output, hidden = rnn_lm(x)

print(f"\nRNN语言模型:")
print(f"  输入: {x.shape}")
print(f"  输出: {output.shape}")
print(f"  隐藏状态: {hidden[0].shape}")
```

```python
class TextDataset(torch.utils.data.Dataset):
    def __init__(self, text, seq_length, vocab_to_idx):
        self.text = text
        self.seq_length = seq_length
        self.vocab_to_idx = vocab_to_idx
        
        tokens = text.split()
        self.indices = [vocab_to_idx.get(t, vocab_to_idx['<UNK>']) for t in tokens]
    
    def __len__(self):
        return max(0, len(self.indices) - self.seq_length)
    
    def __getitem__(self, idx):
        x = torch.tensor(self.indices[idx:idx+self.seq_length])
        y = torch.tensor(self.indices[idx+1:idx+self.seq_length+1])
        return x, y
```

## 4. 语言模型应用

### 4.1 文本生成

```python
def generate_text(model, vocab, start_text, max_length=50, temperature=1.0):
    model.eval()
    
    idx_to_vocab = {i: w for w, i in vocab.items()}
    
    tokens = start_text.split()
    generated = [vocab.get(t, vocab['<UNK>']) for t in tokens]
    
    hidden = None
    
    with torch.no_grad():
        for _ in range(max_length):
            x = torch.tensor([generated[-1]])
            output, hidden = model(x, hidden)
            
            probs = torch.softmax(output[0, 0] / temperature, dim=0)
            next_token = torch.multinomial(probs, 1).item()
            
            generated.append(next_token)
            
            if idx_to_vocab.get(next_token) == '<EOS>':
                break
    
    return ' '.join(idx_to_vocab.get(idx, '<UNK>') for idx in generated)

print("文本生成参数:")
print("  - temperature: 控制随机性")
print("    - 低(0.5): 更确定，保守")
print("    - 高(1.5): 更随机，创造性")
print("  - top-k: 只从概率最高的k个词中采样")
print("  - top-p: 从累积概率达到p的词中采样")
```

### 4.2 文本评估

```python
def evaluate_text_quality(model, text, vocab):
    tokens = text.split()
    indices = [vocab.get(t, vocab['<UNK>']) for t in tokens]
    
    x = torch.tensor([indices[:-1]])
    
    with torch.no_grad():
        output, _ = model(x)
        probs = torch.softmax(output[0], dim=-1)
        
        log_prob = 0
        for i, idx in enumerate(indices[1:]):
            log_prob += torch.log(probs[i, idx] + 1e-10)
        
        perplexity = torch.exp(-log_prob / len(indices[1:]))
    
    return perplexity.item()
```

## 5. 总结

| 模型类型 | 特点 | 优势 | 劣势 |
|---------|------|------|------|
| N-gram | 统计计数 | 简单、快速 | 稀疏性、缺乏泛化 |
| 前馈NN | 固定窗口 | 学习表示 | 窗口限制 |
| RNN LM | 递归结构 | 处理任意长度 | 梯度消失 |
| LSTM/GRU | 门控机制 | 长距离依赖 | 计算复杂 |
| Transformer | 自注意力 | 并行、长距离 | 计算量大 |
