# 文本预处理

## 1. 文本清洗

### 1.1 基本清洗操作

```python
import re
import string

text = "Hello World! This is a NLP tutorial. Visit https://example.com #NLP @user123"

def clean_text(text):
    text = text.lower()
    
    text = re.sub(r'http\S+|www\S+|https\S+', '', text, flags=re.MULTILINE)
    
    text = re.sub(r'@\w+', '', text)
    
    text = re.sub(r'#\w+', '', text)
    
    text = text.translate(str.maketrans('', '', string.punctuation))
    
    text = re.sub(r'\s+', ' ', text).strip()
    
    return text

cleaned = clean_text(text)
print(f"原始文本: {text}")
print(f"清洗后: {cleaned}")
```

### 1.2 处理特殊字符

```python
import unicodedata

def normalize_text(text):
    text = unicodedata.normalize('NFKD', text).encode('ascii', 'ignore').decode('utf-8')
    
    text = re.sub(r'[^\x00-\x7F]+', '', text)
    
    text = re.sub(r'\d+', ' NUM ', text)
    
    return text

text_with_special = "Hello 你好! こんにちは! Price: $100"
normalized = normalize_text(text_with_special)
print(f"原始: {text_with_special}")
print(f"标准化后: {normalized}")
```

def simple_chinese_tokenize(text):
    import re
    pattern = re.compile(r'([\u4e00-\u9fff]+|[a-zA-Z]+|\d+)')

### 2.3 子词分词（Subword Tokenization）

```python
def simple_bpe_example():
    corpus = ["low", "lower", "newest", "widest", "lower", "newest"]
    
    word_freqs = {}
    for word in corpus:
        word_freqs[word] = word_freqs.get(word, 0) + 1
    
    print("词频统计:")
    for word, freq in word_freqs.items():
        print(f"  {word}: {freq}")
    
    vocab = set()
    for word in word_freqs:
        for char in word:
            vocab.add(char)
    
    print(f"\n初始词汇表: {sorted(vocab)}")

simple_bpe_example()
```

### 4.2 词形还原（Lemmatization）

```python
from nltk.stem import WordNetLemmatizer
from nltk.corpus import wordnet

nltk.download('wordnet')
nltk.download('averaged_perceptron_tagger')

lemmatizer = WordNetLemmatizer()

def get_wordnet_pos(word):
    tag = nltk.pos_tag([word])[0][1][0].upper()
    tag_dict = {"J": wordnet.ADJ, "N": wordnet.NOUN, "V": wordnet.VERB, "R": wordnet.ADV}
    return tag_dict.get(tag, wordnet.NOUN)

words = ['running', 'better', 'cats', 'studies', 'worse']

print("词形还原:")
for word in words:
    pos = get_wordnet_pos(word)
    lemma = lemmatizer.lemmatize(word, pos=pos)
    print(f"  {word} -> {lemma} (POS: {pos})")
```

## 5. 文本向量化

### 5.1 One-Hot编码

```python
from sklearn.preprocessing import OneHotEncoder
import numpy as np

corpus = ["I love NLP", "NLP is great", "I love coding"]

vocab = set()
for sentence in corpus:
    vocab.update(sentence.lower().split())
vocab = sorted(vocab)

print(f"词汇表: {vocab}")

word_to_idx = {word: idx for idx, word in enumerate(vocab)}

def one_hot_encode(word, vocab_size):
    vector = np.zeros(vocab_size)
    vector[word_to_idx[word]] = 1
    return vector

print("\nOne-Hot编码示例:")
for word in ['i', 'love', 'nlp']:
    print(f"  {word}: {one_hot_encode(word, len(vocab))}")
```

    print("\nTF-IDF公式:")
    print("  TF(t, d) = 词t在文档d中的出现次数 / 文档d的总词数")
    print("  IDF(t) = log(文档总数 / 包含词t的文档数)")
    print("  TF-IDF(t, d) = TF(t, d) * IDF(t)")

explain_tfidf()

## 6. N-gram模型

```python
from nltk import ngrams

text = "I love natural language processing"
tokens = text.split()

print("N-gram示例:")

for n in [1, 2, 3]:
    n_grams = list(ngrams(tokens, n))
    print(f"\n{n}-grams:")
    for gram in n_grams:
        print(f"  {gram}")

def build_ngram_model(corpus, n=2):
    from collections import defaultdict
    
    ngram_counts = defaultdict(int)
    context_counts = defaultdict(int)
    
    for sentence in corpus:
        tokens = ['<s>'] * (n-1) + sentence.split() + ['</s>']
        for i in range(len(tokens) - n + 1):
            ngram = tuple(tokens[i:i+n])
            context = tuple(tokens[i:i+n-1])
            ngram_counts[ngram] += 1
            context_counts[context] += 1
    
    return ngram_counts, context_counts

corpus = ["I love NLP", "I love coding", "NLP is great"]
ngram_counts, context_counts = build_ngram_model(corpus, n=2)

print("\nBigram模型:")
for ngram, count in ngram_counts.items():
    context = ngram[:-1]
    prob = count / context_counts[context]
    print(f"  P({ngram[-1]}|{context[0]}) = {prob:.4f}")
```

## 7. 完整预处理流程

```python
class TextPreprocessor:
    def __init__(self):
        self.stop_words = set(stopwords.words('english'))
        self.lemmatizer = WordNetLemmatizer()
    
    def clean(self, text):
        text = text.lower()
        text = re.sub(r'http\S+', '', text)
        text = re.sub(r'[^a-zA-Z\s]', '', text)
        text = re.sub(r'\s+', ' ', text).strip()
        return text
    
    def tokenize(self, text):
        return word_tokenize(text)
    
    def remove_stopwords(self, tokens):
        return [t for t in tokens if t not in self.stop_words]
    
    def lemmatize(self, tokens):
        return [self.lemmatizer.lemmatize(t) for t in tokens]
    
    def preprocess(self, text):
        text = self.clean(text)
        tokens = self.tokenize(text)
        tokens = self.remove_stopwords(tokens)
        tokens = self.lemmatize(tokens)
        return tokens

preprocessor = TextPreprocessor()
text = "I love Natural Language Processing! It's amazing. Visit https://nlp.com"
processed = preprocessor.preprocess(text)

print(f"原始文本: {text}")
print(f"预处理后: {processed}")
```
