## 2. Encoder-Decoder架构

### 2.1 基本Encoder

```python
class Encoder(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size, num_layers=2):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        self.rnn = nn.LSTM(embed_size, hidden_size, num_layers, 
                          batch_first=True, bidirectional=True)
        
    def forward(self, x):
        embedded = self.embedding(x)
        outputs, (hidden, cell) = self.rnn(embedded)
        
        return outputs, hidden, cell
```

### 2.2 基本Decoder

```python
class Decoder(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size, num_layers=2):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        self.rnn = nn.LSTM(embed_size, hidden_size, num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_size, vocab_size)
        
    def forward(self, x, hidden, cell):
        embedded = self.embedding(x)
        output, (hidden, cell) = self.rnn(embedded, (hidden, cell))
        prediction = self.fc(output)
        
        return prediction, hidden, cell
```

### 2.3 完整Seq2Seq模型

```python
class Seq2Seq(nn.Module):
    def __init__(self, encoder, decoder, device):
        super().__init__()
        self.encoder = encoder
        self.decoder = decoder
        self.device = device
    
    def forward(self, src, trg, teacher_forcing_ratio=0.5):
        batch_size = src.shape[0]
        trg_len = trg.shape[1]
        trg_vocab_size = self.decoder.fc.out_features
        
        outputs = torch.zeros(batch_size, trg_len, trg_vocab_size).to(self.device)
        
        _, hidden, cell = self.encoder(src)
        
        input = trg[:, 0].unsqueeze(1)
        
        for t in range(1, trg_len):
            output, hidden, cell = self.decoder(input, hidden, cell)
            outputs[:, t] = output.squeeze(1)
            
            teacher_force = torch.rand(1) < teacher_forcing_ratio
            top1 = output.argmax(2)
            input = trg[:, t].unsqueeze(1) if teacher_force else top1
        
        return outputs

print("\nSeq2Seq训练技巧:")
print("  1. Teacher Forcing: 使用真实目标作为输入")
print("  2. Scheduled Sampling: 逐渐减少Teacher Forcing")
print("  3. 梯度裁剪: 防止梯度爆炸")
```

## 3. 注意力机制

### 3.1 Bahdanau注意力

```python
class BahdanauAttention(nn.Module):
    def __init__(self, hidden_size):
        super().__init__()
        
        self.W1 = nn.Linear(hidden_size, hidden_size)
        self.W2 = nn.Linear(hidden_size, hidden_size)
        self.V = nn.Linear(hidden_size, 1)
    
    def forward(self, query, keys, values):
        query = query.unsqueeze(1)
        
        scores = self.V(torch.tanh(self.W1(query) + self.W2(keys)))
        
        attention_weights = torch.softmax(scores, dim=1)
        
        context = torch.sum(attention_weights * values, dim=1)
        
        return context, attention_weights

attention = BahdanauAttention(hidden_size)
```

### 3.2 Luong注意力

```python
class LuongAttention(nn.Module):
    def __init__(self, method='general', hidden_size=256):
        super().__init__()
        
        self.method = method
        self.hidden_size = hidden_size
        
        if method == 'general':
            self.W = nn.Linear(hidden_size, hidden_size)
        elif method == 'concat':
            self.W = nn.Linear(hidden_size * 2, hidden_size)
            self.V = nn.Linear(hidden_size, 1)
    
    def forward(self, query, keys, values):
        if self.method == 'dot':
            scores = torch.bmm(keys, query.unsqueeze(2)).squeeze(2)
        elif self.method == 'general':
            scores = torch.bmm(keys, self.W(query).unsqueeze(2)).squeeze(2)
        else:
            hidden = query.unsqueeze(1).repeat(1, keys.size(1), 1)
            scores = self.V(torch.tanh(self.W(torch.cat([hidden, keys], dim=2)))).squeeze(2)
        
        attention_weights = torch.softmax(scores, dim=1).unsqueeze(1)
        
        context = torch.bmm(attention_weights, values).squeeze(1)
        
        return context, attention_weights

print("\nLuong注意力类型:")
print("  - dot: score = h_t^T * h_s")
print("  - general: score = h_t^T * W * h_s")
print("  - concat: score = v^T * tanh(W * [h_t; h_s])")
```

### 3.3 带注意力的Decoder

```python
class AttnDecoder(nn.Module):
    def __init__(self, vocab_size, embed_size, hidden_size, num_layers=2):
        super().__init__()
        
        self.embedding = nn.Embedding(vocab_size, embed_size)
        self.attention = BahdanauAttention(hidden_size)
        self.rnn = nn.LSTM(hidden_size + embed_size, hidden_size, num_layers, batch_first=True)
        self.fc = nn.Linear(hidden_size * 2, vocab_size)
    
    def forward(self, x, hidden, cell, encoder_outputs):
        embedded = self.embedding(x)
        
        context, attn_weights = self.attention(hidden[-1], encoder_outputs, encoder_outputs)
        
        rnn_input = torch.cat([embedded, context.unsqueeze(1)], dim=2)
        
        output, (hidden, cell) = self.rnn(rnn_input, (hidden, cell))
        
        output = torch.cat([output.squeeze(1), context], dim=1)
        prediction = self.fc(output)
        
        return prediction, hidden, cell, attn_weights

print("\n带注意力的Decoder优势:")
print("  1. 直接访问编码器所有输出")
print("  2. 动态关注相关部分")
print("  3. 缓解长序列问题")
```

## 4. 训练与推理

### 4.1 训练过程

```python
def train_seq2seq(model, train_loader, criterion, optimizer, device, clip=1.0):
    model.train()
    total_loss = 0
    
    for src, trg in train_loader:
        src, trg = src.to(device), trg.to(device)
        
        optimizer.zero_grad()
        
        output = model(src, trg)
        
        output = output[:, 1:].reshape(-1, output.shape[-1])
        trg = trg[:, 1:].reshape(-1)
        
        loss = criterion(output, trg)
        loss.backward()
        
        torch.nn.utils.clip_grad_norm_(model.parameters(), clip)
        
        optimizer.step()
        
        total_loss += loss.item()
    
    return total_loss / len(train_loader)
```

### 4.2 推理过程

```python
def translate(model, src, src_vocab, trg_vocab, max_len=50, device='cpu'):
    model.eval()
    
    src_tensor = torch.tensor([[src_vocab.get(t, src_vocab['<UNK>']) 
                                for t in src.split()]]).to(device)
    
    with torch.no_grad():
        encoder_outputs, hidden, cell = model.encoder(src_tensor)
    
    trg_indices = [trg_vocab['<SOS>']]
    
    for _ in range(max_len):
        trg_tensor = torch.tensor([[trg_indices[-1]]]).to(device)
        
        with torch.no_grad():
            output, hidden, cell, _ = model.decoder(
                trg_tensor, hidden, cell, encoder_outputs
            )
        
        pred_token = output.argmax(2).item()
        trg_indices.append(pred_token)
        
        if pred_token == trg_vocab['<EOS>']:
            break
    
    idx_to_trg = {i: w for w, i in trg_vocab.items()}
    translated = [idx_to_trg[i] for i in trg_indices[1:-1]]
    
    return ' '.join(translated)
```

### 4.3 Beam Search

```python
def beam_search(model, src, src_vocab, trg_vocab, beam_width=5, max_len=50):
    model.eval()
    
    src_tensor = torch.tensor([[src_vocab.get(t, src_vocab['<UNK>']) 
                                for t in src.split()]])
    
    with torch.no_grad():
        encoder_outputs, hidden, cell = model.encoder(src_tensor)
    
    beams = [([trg_vocab['<SOS>']], 0, hidden, cell)]
    completed = []
    
    for _ in range(max_len):
        new_beams = []
        
        for seq, score, hidden, cell in beams:
            if seq[-1] == trg_vocab['<EOS>']:
                completed.append((seq, score))
                continue
            
            trg_tensor = torch.tensor([[seq[-1]]])
            
            with torch.no_grad():
                output, hidden, cell, _ = model.decoder(
                    trg_tensor, hidden, cell, encoder_outputs
                )
            
            probs = torch.log_softmax(output[0, 0], dim=0)
            top_probs, top_indices = probs.topk(beam_width)
            
            for prob, idx in zip(top_probs, top_indices):
                new_seq = seq + [idx.item()]
                new_score = score + prob.item()
                new_beams.append((new_seq, new_score, hidden, cell))
        
        beams = sorted(new_beams, key=lambda x: x[1], reverse=True)[:beam_width]
        
        if all(b[0][-1] == trg_vocab['<EOS>'] for b in beams):
            break
    
    completed.extend(beams)
    best_seq = max(completed, key=lambda x: x[1] / len(x[0]))
    
    idx_to_trg = {i: w for w, i in trg_vocab.items()}
    result = [idx_to_trg[i] for i in best_seq[0][1:-1]]
    
    return ' '.join(result)

print("\nBeam Search参数:")
print(f"  beam_width: 候选数量(通常5-10)")
print(f"  max_len: 最大生成长度")
print(f"  length_penalty: 长度惩罚系数")
```

### 5.1 BLEU

```python
from collections import Counter

def compute_bleu(reference, candidate, n=4):
    def get_ngrams(tokens, n):
        return [tuple(tokens[i:i+n]) for i in range(len(tokens)-n+1)]
    
    ref_tokens = reference.split()
    cand_tokens = candidate.split()
    
    bleu_scores = []
    
    for i in range(1, n+1):
        ref_ngrams = Counter(get_ngrams(ref_tokens, i))
        cand_ngrams = Counter(get_ngrams(cand_tokens, i))
        
        matches = sum((ref_ngrams & cand_ngrams).values())
        total = sum(cand_ngrams.values())
        
        if total == 0:
            bleu_scores.append(0)
        else:
            bleu_scores.append(matches / total)
    
    bp = 1.0
    if len(cand_tokens) < len(ref_tokens):
        bp = np.exp(1 - len(ref_tokens) / len(cand_tokens))
    
    bleu = bp * np.exp(np.mean(np.log(np.array(bleu_scores) + 1e-10)))
    
    return bleu
```
