# 词向量

## 1. 词向量概述

词向量是将词语映射到连续向量空间的技术，能够捕获词语的语义信息。

### 1.1 从One-Hot到分布式表示

```python
import numpy as np

vocab = ['king', 'queen', 'man', 'woman', 'apple', 'orange']
vocab_size = len(vocab)

print("One-Hot编码问题:")
print(f"  向量维度: {vocab_size}")
print(f"  稀疏性: 高度稀疏")
print(f"  语义相似性: 无法表示")

word_to_idx = {word: idx for idx, word in enumerate(vocab)}

def one_hot(word):
    vec = np.zeros(vocab_size)
    vec[word_to_idx[word]] = 1
    return vec

print("\nOne-Hot示例:")
for word in ['king', 'queen']:
    print(f"  {word}: {one_hot(word)}")

print("\n分布式表示优势:")
```

## 2. Word2Vec

### 2.1 CBOW模型

```python
import numpy as np

class SimpleCBOW:
    def __init__(self, vocab_size, embedding_dim):
        self.vocab_size = vocab_size
        self.embedding_dim = embedding_dim
        
        self.W_in = np.random.randn(vocab_size, embedding_dim) * 0.01
        self.W_out = np.random.randn(embedding_dim, vocab_size) * 0.01
    
    def forward(self, context_indices):
        context_vectors = self.W_in[context_indices]
        hidden = np.mean(context_vectors, axis=0)
        
        scores = hidden @ self.W_out
        exp_scores = np.exp(scores - np.max(scores))
        probs = exp_scores / np.sum(exp_scores)
        
        return hidden, probs
    
    def backward(self, context_indices, target_idx, hidden, probs, lr=0.01):
        d_out = probs.copy()
        d_out[target_idx] -= 1
        
        d_W_out = np.outer(hidden, d_out)
        d_hidden = self.W_out @ d_out
        
        d_W_in = np.zeros_like(self.W_in)
        for idx in context_indices:
            d_W_in[idx] = d_hidden / len(context_indices)
        
        self.W_in -= lr * d_W_in
        self.W_out -= lr * d_W_out
    
    def get_embedding(self, word_idx):
        return self.W_in[word_idx]

vocab = ['the', 'cat', 'sat', 'on', 'mat', 'dog']
vocab_size = len(vocab)
word_to_idx = {w: i for i, w in enumerate(vocab)}

cbow = SimpleCBOW(vocab_size, embedding_dim=10)

context = ['the', 'sat', 'on', 'mat']
target = 'cat'
context_indices = [word_to_idx[w] for w in context]
target_idx = word_to_idx[target]

hidden, probs = cbow.forward(context_indices)
print(f"上下文: {context}")
print(f"目标词: {target}")
print(f"预测概率分布: {probs}")
```

```python
class SimpleSkipGram:
    def __init__(self, vocab_size, embedding_dim):
        self.vocab_size = vocab_size
        self.embedding_dim = embedding_dim
        
        self.W_in = np.random.randn(vocab_size, embedding_dim) * 0.01
        self.W_out = np.random.randn(embedding_dim, vocab_size) * 0.01
    
    def forward(self, center_idx):
        hidden = self.W_in[center_idx]
        
        scores = hidden @ self.W_out
        exp_scores = np.exp(scores - np.max(scores))
        probs = exp_scores / np.sum(exp_scores)
        
        return hidden, probs
    
    def backward(self, center_idx, context_idx, hidden, probs, lr=0.01):
        d_out = probs.copy()
        d_out[context_idx] -= 1
        
        d_W_out = np.outer(hidden, d_out)
        d_hidden = self.W_out @ d_out
        
        self.W_in[center_idx] -= lr * d_hidden
        self.W_out -= lr * d_W_out
    
    def get_embedding(self, word_idx):
        return self.W_in[word_idx]
```

### 2.3 负采样

```python
import numpy as np

class NegativeSampling:
    def __init__(self, vocab_size, embedding_dim, num_negative=5):
        self.vocab_size = vocab_size
        self.embedding_dim = embedding_dim
        self.num_negative = num_negative
        
        self.W_in = np.random.randn(vocab_size, embedding_dim) * 0.01
        self.W_out = np.random.randn(vocab_size, embedding_dim) * 0.01
    
    def sigmoid(self, x):
        return 1 / (1 + np.exp(-np.clip(x, -500, 500)))
    
    def sample_negative(self, exclude_idx, word_freqs):
        negatives = []
        while len(negatives) < self.num_negative:
            idx = np.random.choice(self.vocab_size, p=word_freqs)
            if idx != exclude_idx:
                negatives.append(idx)
        return negatives
    
    def train_pair(self, center_idx, context_idx, word_freqs, lr=0.01):
        center_vec = self.W_in[center_idx]
        context_vec = self.W_out[context_idx]
        
        pos_score = self.sigmoid(center_vec @ context_vec)
        pos_grad = (pos_score - 1) * context_vec
        
        neg_indices = self.sample_negative(context_idx, word_freqs)
        neg_grad = np.zeros(self.embedding_dim)
        
        for neg_idx in neg_indices:
            neg_vec = self.W_out[neg_idx]
            neg_score = self.sigmoid(center_vec @ neg_vec)
            neg_grad += neg_score * neg_vec
            
            self.W_out[neg_idx] -= lr * neg_score * center_vec
        
        self.W_in[center_idx] -= lr * (pos_grad + neg_grad)
        self.W_out[context_idx] -= lr * (pos_score - 1) * center_vec

print("负采样优势:")
print("  1. 计算效率高: 只更新少量负样本")
print("  2. 训练稳定: 避免softmax计算")
print("  3. 效果好: 适合大规模语料")
```

## 3. GloVe

### 3.1 GloVe原理

```python
import numpy as np

def build_cooccurrence_matrix(corpus, vocab, window_size=2):
    vocab_size = len(vocab)
    word_to_idx = {w: i for i, w in enumerate(vocab)}
    cooccur = np.zeros((vocab_size, vocab_size))
    
    for sentence in corpus:
        tokens = sentence.split()
        for i, word in enumerate(tokens):
            if word not in word_to_idx:
                continue
            center_idx = word_to_idx[word]
            
            for j in range(max(0, i - window_size), min(len(tokens), i + window_size + 1)):
                if i != j and tokens[j] in word_to_idx:
                    context_idx = word_to_idx[tokens[j]]
                    distance = abs(i - j)
                    cooccur[center_idx, context_idx] += 1.0 / distance
    
    return cooccur

corpus = [
    "the cat sat on the mat",
    "the dog sat on the log",
    "cat and dog played together"
]

vocab = list(set(' '.join(corpus).split()))
cooccur = build_cooccurrence_matrix(corpus, vocab)

print("共现矩阵示例:")
print(f"词汇表: {vocab[:5]}...")
print(f"矩阵形状: {cooccur.shape}")
```

### 3.2 GloVe训练

```python
class SimpleGloVe:
    def __init__(self, vocab_size, embedding_dim):
        self.vocab_size = vocab_size
        self.embedding_dim = embedding_dim
        
        self.W = np.random.randn(vocab_size, embedding_dim) * 0.01
        self.W_context = np.random.randn(vocab_size, embedding_dim) * 0.01
        self.b = np.zeros(vocab_size)
        self.b_context = np.zeros(vocab_size)
    
    def weighting_function(self, x, x_max=100, alpha=0.75):
        if x < x_max:
            return (x / x_max) ** alpha
        return 1.0
    
    def train(self, cooccur, epochs=100, lr=0.01):
        for epoch in range(epochs):
            total_loss = 0
            
            for i in range(self.vocab_size):
                for j in range(self.vocab_size):
                    if cooccur[i, j] == 0:
                        continue
                    
                    weight = self.weighting_function(cooccur[i, j])
                    diff = self.W[i] @ self.W_context[j] + self.b[i] + self.b_context[j] - np.log(cooccur[i, j])
                    
                    loss = weight * diff ** 2
                    total_loss += loss
                    
                    grad = weight * diff
                    
                    self.W[i] -= lr * grad * self.W_context[j]
                    self.W_context[j] -= lr * grad * self.W[i]
                    self.b[i] -= lr * grad
                    self.b_context[j] -= lr * grad
            
            if (epoch + 1) % 20 == 0:
                print(f"Epoch {epoch+1}, Loss: {total_loss:.4f}")
    
    def get_embedding(self, word_idx):
        return self.W[word_idx] + self.W_context[word_idx]

print("GloVe特点:")
print("  1. 利用全局统计信息")
print("  2. 结合了矩阵分解和局部上下文窗口")
print("  3. 训练效率高")
```

## 4. 词向量评估

### 4.1 相似度评估

```python
def cosine_similarity(v1, v2):
    return np.dot(v1, v2) / (np.linalg.norm(v1) * np.linalg.norm(v2))

def evaluate_embeddings(embeddings, word_pairs, similarities):
    predicted = []
    for w1, w2 in word_pairs:
        if w1 in embeddings and w2 in embeddings:
            sim = cosine_similarity(embeddings[w1], embeddings[w2])
            predicted.append(sim)
    
    from scipy.stats import spearmanr
    correlation, _ = spearmanr(predicted, similarities)
    return correlation

np.random.seed(42)
```

### 4.2 类比任务

```python
def analogy_task(embeddings, a, b, c):
    """
    a is to b as c is to ?
    """
    if not all(w in embeddings for w in [a, b, c]):
        return None
    
    target = embeddings[b] - embeddings[a] + embeddings[c]
    
    best_word = None
    best_sim = -1
    
    for word, vec in embeddings.items():
        if word in [a, b, c]:
            continue
        sim = cosine_similarity(target, vec)
        if sim > best_sim:
            best_sim = sim
            best_word = word
    
    return best_word, best_sim

print("\n类比任务示例:")
print("  king - man + woman = queen")
print("  paris - france + italy = rome")
print("  walked - walk + run = ran")
```

## 6. 总结

| 方法 | 原理 | 优势 | 劣势 |
|------|------|------|------|
| Word2Vec CBOW | 上下文预测中心词 | 训练快 | 忽略词序 |
| Word2Vec Skip-gram | 中心词预测上下文 | 稀有词效果好 | 训练慢 |
| GloVe | 全局共现矩阵 | 全局信息 | 需要大内存 |
| FastText | 子词嵌入 | 处理未登录词 | 向量维度增加 |
