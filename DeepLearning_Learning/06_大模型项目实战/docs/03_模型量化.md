## 2. 量化方法

### 2.1 量化基础

```python
import numpy as np

class Quantizer:
    def __init__(self, bits=8):
        self.bits = bits
        self.qmin = -(2 ** (bits - 1))
        self.qmax = 2 ** (bits - 1) - 1
    
    def quantize(self, x, scale, zero_point):
        x_q = x / scale + zero_point
        x_q = np.clip(x_q, self.qmin, self.qmax)
        x_q = np.round(x_q).astype(np.int8)
        return x_q
    
    def dequantize(self, x_q, scale, zero_point):
        return (x_q - zero_point) * scale
    
    def compute_scale_zp(self, x):
        rmin, rmax = x.min(), x.max()
        
        scale = (rmax - rmin) / (self.qmax - self.qmin)
        zero_point = self.qmin - rmin / scale
        
        return scale, zero_point

print("量化公式:")
print("  量化: x_q = round(x / scale + zero_point)")
print("  反量化: x = (x_q - zero_point) * scale")

quantizer = Quantizer(bits=8)
x = np.array([0.1, 0.5, 1.0, -0.3, -1.5])
scale, zp = quantizer.compute_scale_zp(x)
x_q = quantizer.quantize(x, scale, zp)
x_dq = quantizer.dequantize(x_q, scale, zp)

print(f"\n原始值: {x}")
print(f"量化后: {x_q}")
print(f"反量化: {x_dq}")
print(f"误差: {np.abs(x - x_dq).mean():.6f}")
```

def dynamic_quantize_linear(weight):
    rmin, rmax = weight.min(), weight.max()
    scale = (rmax - rmin) / 255
    zero_point = 128 - rmax / scale
    
    weight_q = torch.clamp(torch.round(weight / scale + zero_point), 0, 255).to(torch.uint8)
    
    return weight_q, scale, zero_point

### 2.4 量化感知训练

```python
class FakeQuantize(torch.autograd.Function):
    @staticmethod
    def forward(ctx, x, scale, zero_point, qmin, qmax):
        x_q = torch.clamp(torch.round(x / scale + zero_point), qmin, qmax)
        x_dq = (x_q - zero_point) * scale
        return x_dq
    
    @staticmethod
    def backward(ctx, grad_output):
        return grad_output, None, None, None, None

class QuantizeLinear(nn.Module):
    def __init__(self, in_features, out_features, bits=8):
        super().__init__()
        
        self.weight = nn.Parameter(torch.randn(out_features, in_features))
        self.bias = nn.Parameter(torch.zeros(out_features))
        
        self.bits = bits
        self.qmin = -(2 ** (bits - 1))
        self.qmax = 2 ** (bits - 1) - 1
        
        self.weight_scale = nn.Parameter(torch.ones(1))
        self.weight_zp = nn.Parameter(torch.zeros(1))
        
        self.act_scale = nn.Parameter(torch.ones(1))
        self.act_zp = nn.Parameter(torch.zeros(1))
    
    def forward(self, x):
        if self.training:
            w_q = FakeQuantize.apply(
                self.weight, self.weight_scale, self.weight_zp,
                self.qmin, self.qmax
            )
            x_q = FakeQuantize.apply(
                x, self.act_scale, self.act_zp,
                self.qmin, self.qmax
            )
        else:
            w_q = torch.clamp(
                torch.round(self.weight / self.weight_scale + self.weight_zp),
                self.qmin, self.qmax
            )
            w_q = (w_q - self.weight_zp) * self.weight_scale
        
        return nn.functional.linear(x, w_q, self.bias)

print("\n量化感知训练 (QAT):")
print("  - 训练时模拟量化效果")
print("  - 梯度直通估计器 (STE)")
print("  - 使模型适应量化误差")
print("  - 精度损失最小")
```

class GPTQQuantizer:
    def __init__(self, bits=4, group_size=128):
        self.bits = bits
        self.group_size = group_size
    
    def quantize_weight(self, weight):
        out_features, in_features = weight.shape
        
        weight_q = torch.zeros_like(weight)
        
        for i in range(0, in_features, self.group_size):
            group = weight[:, i:i+self.group_size]
            
            qmin = -(2 ** (self.bits - 1))
            qmax = 2 ** (self.bits - 1) - 1
            
            scale = group.abs().max() / qmax
            group_q = torch.clamp(torch.round(group / scale), qmin, qmax)
            
            weight_q[:, i:i+self.group_size] = group_q
        
        return weight_q

print("\nGPTQ特点:")
print("  - 分组量化 (Group Quantization)")
print("  - 基于Hessian矩阵的优化")
print("  - 量化后精度损失小")

class NF4Quantizer:
    def __init__(self):
        self.quantile_values = self._compute_nf4_quantiles()
    
    def _compute_nf4_quantiles(self):
        from scipy.stats import norm
        n = 16
        quantiles = norm.ppf(torch.linspace(0, 1, n + 1)[1:])
        return quantiles
    
    def quantize(self, x):
        abs_x = x.abs()
        max_val = abs_x.max()
        normalized = x / max_val
        
        indices = torch.searchsorted(self.quantile_values.to(x.device), normalized)
        return indices, max_val
    
    def dequantize(self, indices, max_val):
        return self.quantile_values[indices] * max_val
