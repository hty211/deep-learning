## 3. PEFT方法详解

### 3.1 Adapter

```python
import torch
import torch.nn as nn

class Adapter(nn.Module):
    def __init__(self, hidden_size, adapter_size=64, dropout=0.1):
        super().__init__()
        
        self.down_proj = nn.Linear(hidden_size, adapter_size)
        self.up_proj = nn.Linear(adapter_size, hidden_size)
        self.dropout = nn.Dropout(dropout)
        self.activation = nn.GELU()
        
        nn.init.zeros_(self.up_proj.weight)
        nn.init.zeros_(self.up_proj.bias)
    
    def forward(self, x):
        residual = x
        x = self.down_proj(x)
        x = self.activation(x)
        x = self.dropout(x)
        x = self.up_proj(x)
        return residual + x

print("Adapter方法:")
print("  - 在Transformer层中插入小型瓶颈网络")
print("  - 下投影 -> 激活 -> 上投影")
print("  - 参数量: hidden_size * adapter_size * 2")
print("  - 初始化为零，训练开始时等效于恒等映射")
```

### 3.2 Prefix Tuning

```python
class PrefixTuning(nn.Module):
    def __init__(self, num_layers, hidden_size, prefix_length=10):
        super().__init__()
        
        self.prefix_length = prefix_length
        
        self.prefix_embeddings = nn.Parameter(
            torch.randn(prefix_length, hidden_size) * 0.02
        )
        
        self.prefix_mlp = nn.ModuleList([
            nn.Sequential(
                nn.Linear(hidden_size, hidden_size),
                nn.Tanh()
            ) for _ in range(num_layers)
        ])
    
    def forward(self, hidden_states, layer_idx):
        prefix = self.prefix_mlp[layer_idx](self.prefix_embeddings)
        prefix = prefix.unsqueeze(0).expand(hidden_states.size(0), -1, -1)
        return torch.cat([prefix, hidden_states], dim=1)

print("\nPrefix Tuning:")
print("  - 在输入前添加可学习的前缀向量")
print("  - 每层都有独立的前缀参数")
print("  - 参数量: prefix_length * hidden_size * num_layers")
print("  - 适合生成任务")
```

### 3.3 Prompt Tuning

```python
class PromptTuning(nn.Module):
    def __init__(self, vocab_size, hidden_size, num_virtual_tokens=20):
        super().__init__()
        
        self.num_virtual_tokens = num_virtual_tokens
        
        self.virtual_tokens = nn.Parameter(
            torch.randn(num_virtual_tokens, hidden_size) * 0.02
        )
    
    def forward(self, input_embeds):
        batch_size = input_embeds.size(0)
        virtual_tokens = self.virtual_tokens.unsqueeze(0).expand(batch_size, -1, -1)
        return torch.cat([virtual_tokens, input_embeds], dim=1)

print("\nPrompt Tuning:")
print("  - 只在输入层添加可学习的虚拟token")
print("  - 参数量最少: num_virtual_tokens * hidden_size")
print("  - 简单但效果可能不如其他方法")
```

### 3.4 LoRA

```python
class LoRALayer(nn.Module):
    def __init__(self, in_features, out_features, rank=8, alpha=16):
        super().__init__()
        
        self.rank = rank
        self.alpha = alpha
        self.scaling = alpha / rank
        
        self.lora_A = nn.Parameter(torch.randn(in_features, rank) * 0.01)
        self.lora_B = nn.Parameter(torch.zeros(rank, out_features))
    
    def forward(self, x, original_output):
        lora_output = (x @ self.lora_A @ self.lora_B) * self.scaling
        return original_output + lora_output

print("\nLoRA (Low-Rank Adaptation):")
print("  - 在权重矩阵旁添加低秩分解")
print("  - W' = W + BA, 其中B和A是低秩矩阵")
print("  - 参数量: rank * (in_features + out_features)")
print("  - 训练时只更新LoRA参数")
print("  - 推理时可以合并到原权重")
```

def selection_guide():
